/* This code is used to calculate the basic salary of an employee based on the condition: 

1. If clerk house rent allowances=0 else 8%. 

2. If basic salary <= 300, dearness allowence = 70% else 85%. 

3. Total salary is sum of all these. */
#include <stdio.h>

void main() {

    float hra, da, total_salary; 

    int bs;

    char clerk;

    // Input basic salary and designation

    printf("Enter Basic Salary: ");

    scanf("%d", &bs);

    getchar(); // To consume the newline character left by previous scanf

    printf("Enter 'y' if you area a clerk else 'n': ");

    scanf("%c", &clerk);

    if (bs <= 300) {

        da = (bs*70)/100; // Clerk gets 0 HRA

    } else {

        da = (bs*85)/100; // 8% of basic salary for other employees

    }

    if (clerk =='y') {

        hra=0; // 70% of basic salary if salary is less than 3000

    } else if(clerk=='n') {

        hra=(bs*8)/100; // 85% of basic salary if salary is more than or equal to 3000

    }

    total_salary = bs + hra + da;

    // Output the total salary

    printf("Total Salary: %.2f\n", total_salary);
    
}

/* This code is used to calculate the sum of series: 1 + 3/2! + 5/3! + 7/4! +...........  up to n terms. */
#include <stdio.h>

void main() { 
    int n;

    double sum = 0.0, fact = 1.0;

    printf("Enter the number of terms up to which you want to sum up: ");

    scanf("%d", &n);

    for (int i = 1; i <= n; i++) {

        fact *= i;  // Calculate factorial

        sum += (2.0 * i - 1) / fact;  // Add the term to the sum

    }

    printf("The sum of the series up to %d terms: %lf\n", n, sum);
    
}
 

    
/* This code is used to check whethe a number is Armstrong number or not. */
#include <stdio.h>

void main() {
   int num, actual, remainder, result = 0, n = 0, power;

    printf("Enter a number: ");

    scanf("%d", &num);

    actual = num;

    while (actual != 0) {

        actual /= 10;

        ++n;

    }

    actual = num;

    while (actual != 0) {

        remainder = actual % 10;

        power = 1;

        for (int i = 1; i <= n; i++) {

            power *= remainder;

        }

        result += power;

        actual /= 10;

    }

    if (result == num) {

        printf("%d is an Armstrong number.\n", num);

    } else {

        printf("%d is not an Armstrong number.\n", num);

    }
    
}
   
  
   
#include <stdio.h>
void fibonacci(int n) {

    int first = 1, second = 1, next;

    printf("Fibonacci Series: %d %d", first, second);

    for (int i = 3; i <= n; i++) {

        next = first + second;

        printf(" %d", next);

        first = second;

        second = next;

    }

    printf("\n");

}

/* This code is used for addition of two matrices.*/
#include <stdio.h>

int main() {

    int row, col, i, j;

    

    // Input matrix dimensions

    printf("Enter the number of rows and columns of the matrices: ");

    scanf("%d %d", &row, &col);


    int matrix1[row][col], matrix2[row][col], sum[row][col];


    // Input elements of first matrix

    printf("Enter elements of first matrix:\n");

    for (i = 0; i < row; i++) {

        for (j = 0; j < col; j++) {

            scanf("%d", &matrix1[i][j]);

        }

    }


    // Input elements of second matrix

    printf("Enter elements of second matrix:\n");

    for (i = 0; i < row; i++) {

        for (j = 0; j < col; j++) {

            scanf("%d", &matrix2[i][j]);

        }

    }


    // Addition of matrices using for(;;) loop

    i = 0;

    for (;;) {

        if (i >= row) break; // Exit condition

        j = 0;

        for (;;) {

            if (j >= col) break; // Exit condition

            sum[i][j] = matrix1[i][j] + matrix2[i][j]; // Add elements

            j++;

        }

        i++;

    }


    // Displaying the result matrix

    printf("Sum of matrices:\n");

    for (i = 0; i < row; i++) {

        for (j = 0; j < col; j++) {

            printf("%d ", sum[i][j]);

        }

        printf("\n");

    }


    return 0;

}

   
  
   
/*This code is used for multiplication of two matraice A ans B. */
#include <stdio.h>

void main() {

   int row1, col1, row2, col2, i, j, k;

    printf("Enter rows and columns of first matrix: ");

    scanf("%d %d", &row1, &col1);

    printf("Enter rows and columns of second matrix: ");

    scanf("%d %d", &row2, &col2);

    if (col1 != row2) {

        printf("Matrix multiplication not possible!\n");

    }

    int matrix1[row1][col1], matrix2[row2][col2], result[row1][col2];

    printf("Enter elements of first matrix:\n");

    for (i = 0; i < row1; i++)

        for (j = 0; j < col1; j++)

            scanf("%d", &matrix1[i][j]);

    printf("Enter elements of second matrix:\n");

    for (i = 0; i < row2; i++)

        for (j = 0; j < col2; j++)

            scanf("%d", &matrix2[i][j]);


    for (i = 0; i < row1; i++)

        for (j = 0; j < col2; j++) {

            result[i][j] = 0;

            for (k = 0; k < col1; k++)

                result[i][j] += matrix1[i][k] * matrix2[k][j];

        }


    printf("Resultant Matrix:\n");

    for (i = 0; i < row1; i++) {

        for (j = 0; j < col2; j++)

            printf("%d ", result[i][j]);

        printf("\n");

    }
    
}
